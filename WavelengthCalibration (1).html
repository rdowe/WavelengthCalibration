X_coord = X.astype(float)

# Use this parameter to adjust the continuum level if data are not bias subtracted
contin = 0

# Compound astropy model with constant value plus a whole bunch of gaussians
g_init=models.Const1D(amplitude=contin)

for j in peak_locs :
    g_init = g_init + models.Gaussian1D(amplitude=lamp_sum[j], mean=X_coord[j], stddev=1.6)

# Perform fit using a least squares fitter
fitter = fitting.LevMarLSQFitter()
